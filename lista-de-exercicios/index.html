LOGICA PROGRAMAÇÃO - EXERCICIO 2

const booleano1 = true const booleano3 = true (!booleano2)
const booleano2 = false const booleano4 = false (!booleano3)

a) `booleano1 && booleano2 && !booleano4`
resposta: `true e false e flase`

b) `(booleano1 && booleano2) || !booleano3`
resposta: true e true ou true

c) `(booleano2 || booleano3) && (booleano4 || booleano1)`
resposta: (false ou true) && (false ou true)

d) `!(booleano2 && booleano3) || !(booleano1 && booleano3)`
resposta: (true e false ) ou (false e false)

e) `!(booleano1) && !(booleano3) || (!booleano4 && booleano3 && booleano3)`
